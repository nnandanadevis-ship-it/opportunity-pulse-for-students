Skill-Based Match Algorithm
// Node.js — runs on /api/opportunities function computeMatch(user, opp) { const userSet = new Set( user.skills.map(s => s.toLowerCase()) ); const required = opp.required_skills || []; const hits = required.filter( s => userSet.has(s.toLowerCase()) ); const base = required.length ? Math.round((hits.length / required.length) * 70) : 50; // bonus: domain match (arts/sports/tech) const domainBonus = user.skill_domains.some( d => opp.skill_categories?.includes(d) ) ? 20 : 0; return Math.min(base + domainBonus + 10, 100); }
